% Structuring a talk is a difficult task and the following structure
% may not be suitable. Here are some rules that apply for this
% solution: 

% - Exactly two or three sections (other than the summary).
% - At *most* three subsections per section.
% - Talk about 30s to 2min per frame. So there should be between about
%   15 and 30 frames, all told.

% - A conference audience is likely to know very little of what you
%   are going to talk about. So *simplify*!
% - In a 20min talk, getting the main ideas across is hard
%   enough. Leave out details, even if it means being less precise than
%   you think necessary.
% - If you omit details that are vital to the proof/implementation,
%   just say so once. Everybody will be happy with that.

\begin{frame}{Useful Terms}{Base Rules}

    \end{itemize}

\begin{frame}{Flags}{Base Rules}
  \begin{itemize}
  \item You want these
    \subitem{Each has some point value}
  \item Restrictions on placement
    \subitem{Supervised by judges}
  \item Carried flags dispel all items
    \subitem{They remain dispelled for one minute}
  \item No flags within 10 feet of glyphs
    \subitem{If dropped, glyph dispelled}
  \end{itemize}
\end{frame}

\begin{frame}{Capture}{Base Rules}
  \begin{itemize}
  \item Tag enemy in your home territory
  \item Drop flags
  \item Take nonconcealable stuff
     \subitem{And known concealed}
  \item Escort to Jail (all the way)
  \item If stunned/captured, captive freed
  \end{itemize}
\end{frame}

\begin{frame}{Jail}{Base Rules}
  \begin{itemize}
  \item Touch Glyph, possibly via chains
  \item Jailbreak every 15 minutes
    \begin{itemize}
    \item Free backs (jazz hands!)
    \end{itemize}
  \item Free teammate, receive free backs
  \item Jailer Fedora
    \begin{itemize}
    \item Not necessary for functioning jail
    \item Can relax chain requirement
    \item Truth serum
      \item Pinky finger
      \item Six Y/N questions, then freed
    \end{itemize}
  \end{itemize}
\end{frame}
