         % back arm
         \fill[yshift=\eparmjoint*0.5cm,rotate=\epbarm] (0,-\eparmlength) ellipse ({\eparmwidth} and \eparmlength);

            \useasboundingbox (0, 0);
            \if\epbwand* \else

               \fill[color=black,yshift=\eparmjoint*0.5cm,rotate=\epbarm,yshift=(-2*\eparmlength+\eparmjointoffset)*0.5cm+0.3cm,rotate=-\epbarm,rotate=\epbwandangle]

            \fi
         \end{scope}
      \fi


      %body
      \fill[line width=2pt,draw=white] (0,0) ellipse (1 and \epbodyheight);

      %head

         \fill[line width=2pt,draw=white,yshift=(-\epbodyheight-0.2*\eplegjointoffset)*0.5cm,rotate=60]

      \else
         \if\epcenterleg* \else

            \fill[line width=2pt,draw=white] (0,-\epbodyheight-\eplegheight-\epbodyspacing) ellipse ({\eplegwidth} and \eplegheight);

         \fi
      \fi
         
      %belt

      \fi

            \fill[color=black,yshift=\eparmjoint*0.5cm,rotate=\epfarm,yshift=(-2*\eparmlength+\eparmjointoffset)*0.5cm+0.3cm,rotate=-\epfarm,rotate=\epfwandangle]

         \fi
         \if\epfflag* \else
            %front flag
            \fill[color=\epfflag,yshift=\eparmjoint*0.5cm,rotate=\epfarm,yshift=(-2*\eparmlength+\eparmjointoffset)*0.5cm+0.3cm,rotate=-\epfarm,rotate=-90,rotate=\epfflagangle]
